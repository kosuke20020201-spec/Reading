// ── Drawing ──────────────────────────────────── let drawingCanvas, drawingCtx, isDrawing = false, lastX = 0, lastY = 0, canvasRect = null, dpiRatio = 1; function openDrawingModal() { document.getElementById('drawingOverlay').classList.add('open'); setTimeout(() => { drawingCanvas = document.getElementById('drawingCanvas'); drawingCtx = drawingCanvas.getContext('2d', { willReadFrequently: false }); resizeCanvasForDPI(); drawingHistory = []; saveDrawingState(); drawingCtx.fillStyle = 'white'; drawingCtx.fillRect(0, 0, drawingCanvas.width, drawingCanvas.height); setupDrawingListeners(); }, 10); } function resizeCanvasForDPI() { const rect = drawingCanvas.getBoundingClientRect(); dpiRatio = window.devicePixelRatio || 1; // Set internal resolution to match physical pixels drawingCanvas.width = rect.width * dpiRatio; drawingCanvas.height = rect.height * dpiRatio; // Scale context to match DPI drawingCtx.scale(dpiRatio, dpiRatio); // Set display size (CSS pixels) drawingCanvas.style.width = rect.width + 'px'; drawingCanvas.style.height = rect.height + 'px'; // Cache rect for coordinate calculations canvasRect = rect; } function closeDrawingModal() { document.getElementById('drawingOverlay').classList.remove('open'); isDrawing = false; } function setDrawingMode(mode) { drawingMode = mode; document.querySelectorAll('.drawing-tool-btn').forEach(b => b.classList.remove('active')); event.target.classList.add('active'); } function setupDrawingListeners() { const canvas = drawingCanvas; const getCoordinates = (clientX, clientY) => { // Always recalculate rect for touch devices const rect = canvas.getBoundingClientRect(); return { x: clientX - rect.left, y: clientY - rect.top }; }; const drawLine = (fromX, fromY, toX, toY) => { if (drawingMode === 'draw') { drawingCtx.lineWidth = 2; drawingCtx.lineCap = 'round'; drawingCtx.lineJoin = 'round'; drawingCtx.strokeStyle = document.getElementById('brushColor').value; drawingCtx.beginPath(); drawingCtx.moveTo(fromX, fromY); drawingCtx.lineTo(toX, toY); drawingCtx.stroke(); } else if (drawingMode === 'erase') { drawingCtx.clearRect(toX - 10, toY - 10, 20, 20); } }; // ポインターイベント（マウス対応） canvas.addEventListener('pointerdown', (e) => { if (e.pointerType !== 'mouse') return; // タッチはタッチイベントで処理 isDrawing = true; saveDrawingState(); const coords = getCoordinates(e.clientX, e.clientY); lastX = coords.x; lastY = coords.y; }, { passive: false }); canvas.addEventListener('pointermove', (e) => { if (e.pointerType !== 'mouse' || !isDrawing) return; e.preventDefault(); const coords = getCoordinates(e.clientX, e.clientY); drawLine(lastX, lastY, coords.x, coords.y); lastX = coords.x; lastY = coords.y; }, { passive: false }); canvas.addEventListener('pointerup', (e) => { if (e.pointerType === 'mouse') isDrawing = false; }, { passive: false }); canvas.addEventListener('pointerleave', (e) => { if (e.pointerType === 'mouse') isDrawing = false; }, { passive: false }); // タッチイベント（スマートフォン・タブレット対応） canvas.addEventListener('touchstart', (e) => { e.preventDefault(); isDrawing = true; saveDrawingState(); const touch = e.touches[0]; const coords = getCoordinates(touch.clientX, touch.clientY); lastX = coords.x; lastY = coords.y; }, { passive: false }); canvas.addEventListener('touchmove', (e) => { e.preventDefault(); if (!isDrawing || !e.touches[0]) return; const touch = e.touches[0]; const coords = getCoordinates(touch.clientX, touch.clientY); drawLine(lastX, lastY, coords.x, coords.y); lastX = coords.x; lastY = coords.y; }, { passive: false }); canvas.addEventListener('touchend', (e) => { e.preventDefault(); isDrawing = false; }, { passive: false }); canvas.addEventListener('touchcancel', () => { isDrawing = false; }); } function saveDrawingState() { drawingHistory.push(drawingCanvas.toDataURL()); } function undoDrawing() { if (drawingHistory.length > 1) { drawingHistory.pop(); const img = new Image(); img.onload = () => { drawingCtx.clearRect(0, 0, drawingCanvas.width, drawingCanvas.height); drawingCtx.drawImage(img, 0, 0); }; img.src = drawingHistory[drawingHistory.length - 1]; } } function clearDrawing() { if (confirm('スケッチをクリアしますか？')) { drawingHistory = []; saveDrawingState(); drawingCtx.fillStyle = 'white'; drawingCtx.fillRect(0, 0, drawingCanvas.width, drawingCanvas.height); } } function saveDrawing() { const dataUrl = drawingCanvas.toDataURL('image/png'); const gallery = document.getElementById('drawingGallery'); const thumb = document.createElement('div'); thumb.className = 'drawing-thumb'; thumb.innerHTML = ` `; gallery.appendChild(thumb); closeDrawingModal(); showToast('🎨 スケッチを保存しました'); } function deleteDrawing(idx) { const gallery = document.getElementById('drawingGallery'); const thumbs = gallery.querySelectorAll('.drawing-thumb'); if (thumbs[idx]) thumbs[idx].remove(); } function openDrawingViewer(src) { const viewer = document.createElement('div'); viewer.style.cssText = 'position:fixed;inset:0;background:rgba(0,0,0,0.8);z-index:2000;display:flex;align-items:center;justify-content:center;padding:1rem;'; viewer.innerHTML = ``; document.body.appendChild(viewer); }
