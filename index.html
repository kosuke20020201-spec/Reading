const SHEET_NAME = 'entries'; function doGet(e) { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME); const rows = sheet.getDataRange().getValues(); const headers = rows[0]; const entries = rows.slice(1).map(row => { const obj = {}; headers.forEach((h, i) => { if (h === 'tags') { obj[h] = row[i] ? row[i].split(',') : []; } else { obj[h] = row[i]; } }); return obj; }).filter(e => e.id); return ContentService.createTextOutput(JSON.stringify(entries)) .setMimeType(ContentService.MimeType.JSON); } function doPost(e) { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME); const data = JSON.parse(e.postData.contents); const rows = sheet.getDataRange().getValues(); const headers = rows[0]; // 既存データをIDをキーにしたMapに変換 const existingIds = new Map(); for (let i = 1; i < rows.length; i++) { const id = rows[i][headers.indexOf('id')]; if (id) existingIds.set(String(id), i + 1); } data.forEach(entry => { const row = headers.map(h => { if (h === 'tags') return (entry[h] || []).join(','); return entry[h] ?? ''; }); const rowNum = existingIds.get(String(entry.id)); if (rowNum) { sheet.getRange(rowNum, 1, 1, row.length).setValues([row]); } else { sheet.appendRow(row); } }); return ContentService.createTextOutput(JSON.stringify({ status: 'ok' })) .setMimeType(ContentService.MimeType.JSON); }
